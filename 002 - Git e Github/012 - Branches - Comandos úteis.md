```bash

git pull
# baixa as alterações e faz uma junção (merge)

# baixa apenas as alterações sem fazer o merge
git fetch

# clonar apenas uma branch
git clone link-do-repositorio --branch nome-da-branch --sigle-branch

# remove as alterações
git stash

# listando as alterações
git stash list

# excluíndo as últimas alterações da lista
git stash pop

# aplicando as alterações
git stash apply

```
